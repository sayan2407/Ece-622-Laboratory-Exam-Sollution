1)Write a MATLAB program to compute the convolution sum of a rectangle signal (or gate function) with itself for N=5
x(n)=rect(n/2N)=(n/2N)

clc;
clear all;
close all;
x=ones(1,11);
h=ones(1,11);
nx=[-5:5];
LEx=nx(1);
nh=[-5:5];
LEh=nh(1);
Nx=length(x);
Nh=length(h);
LEy=LEx+LEh;
REx=nx(Nx);
REh=nh(Nh);
REy=REx+REh;
Ny=Nx+Nh-1;
Xn=[x zeros(1,Ny-Nx)];
Hn=[h zeros(1,Ny-Nh)];
for i=0 : Ny-1
    sum=0;
    for j=0:i
        sum=sum+Xn(j+1) * Hn (i-j+1);
    end
    Y(i+1)=sum;
end
subplot(121);
stem(nx,x,'filled');

xlabel('n');
title('x(n) & h(n)');
subplot(122);
ny=LEy:REy;
stem(ny,Y,'filled');
xlabel('n');
title('y(n)');
